    def get_raw_dataframe(self) -> pd.DataFrame:
    """
    Get the raw report data as a pandas DataFrame.

    Returns
    -------
    pd.DataFrame
        The raw report data.
    """
    if not self.report:
        raise ValueError("Report data is not available")
    return pd.DataFrame(self.report)
